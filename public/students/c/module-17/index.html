1. Basic Calculator Using Functions

Performs +, –, ×, ÷ using user-defined functions.

float add(float a, float b){ return a + b; }
float sub(float a, float b){ return a - b; }
float mul(float a, float b){ return a * b; }
float divi(float a, float b){ return a / b; }

int main() {
    float x, y;
    char op;
    printf("Enter a operator (+ - * /): ");
    scanf("%c", &op);
    printf("Enter two numbers: ");
    scanf("%f %f", &x, &y);

    switch(op){
        case '+': printf("%f", add(x,y)); break;
        case '-': printf("%f", sub(x,y)); break;
        case '*': printf("%f", mul(x,y)); break;
        case '/': printf("%f", divi(x,y)); break;
        default : printf("Invalid");
    }
}

2. Student Result / Management System (File Handling)

Stores student records in a text file.

struct Student {
    int roll;
    char name[20];
    float marks;
};

void addRecord() {
    FILE *fp = fopen("students.txt", "a");
    struct Student s;
    printf("Enter roll, name, marks: ");
    scanf("%d %s %f", &s.roll, s.name, &s.marks);
    fprintf(fp, "%d %s %f\n", s.roll, s.name, s.marks);
    fclose(fp);
}

void displayRecords() {
    FILE *fp = fopen("students.txt", "r");
    struct Student s;
    while(fscanf(fp, "%d %s %f", &s.roll, s.name, &s.marks) != EOF){
        printf("%d %s %f\n", s.roll, s.name, s.marks);
    }
    fclose(fp);
}

4. Contact Management System (Structures + Files)

Add / search / delete contacts using files.

struct Contact {
    char name[30];
    char phone[15];
};

void addContact() {
    FILE *fp = fopen("contacts.txt", "a");
    struct Contact c;

    printf("Enter name & phone: ");
    scanf("%s %s", c.name, c.phone);

    fprintf(fp, "%s %s\n", c.name, c.phone);
    fclose(fp);
}

void searchContact(char *key) {
    FILE *fp = fopen("contacts.txt", "r");
    struct Contact c;

    while(fscanf(fp, "%s %s", c.name, c.phone) != EOF) {
        if(strcmp(c.name, key) == 0){
            printf("Found: %s %s\n", c.name, c.phone);
            fclose(fp);
            return;
        }
    }
    printf("Not Found\n");
    fclose(fp);
}

6. Number Guessing Game

Random number guessing with attempts.

int main() {
    int secret, guess;
    srand(time(0));
    secret = rand() % 100 + 1;

    do {
        printf("Guess number (1-100): ");
        scanf("%d", &guess);

        if(guess > secret) printf("Too high!\n");
        else if(guess < secret) printf("Too low!\n");
    } while(guess != secret);

    printf("Correct!");
}
